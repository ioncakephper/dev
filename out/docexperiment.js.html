const { isEmpty, isString, isObject, isArray, isUndefined } = require("lodash");


/**
 * A list of the possible results that a dialog can return.
 * @const
 * @type {number}           
 */
const DIALOG_RESULTS = {
    Cancel: 0,
    Ok: 1,
    No: 2,
    Yes: 3,
}

/**
 * @typedef StandardItem
 * @description internal representation of a sidebar item
 * @type {object}
 * @property {string} label - string used as label.
 * @property {Array<string|StandardItem>} items - the items inside.
 */

/**
 * Takes in an object and converts it to a standard item object.       
 * @param {object} item - the object to convert to a standard item object.       
 * @returns {StandardItem} - the standard item object.       
 */
function buildStandardItem(item) {
    if (isString(item)) {
        item = {
            label: item.trim()
        }
    }
    if (isObject(item)) {
        if (!isStandardItem(item)) {
            item = convertObjectToStandardItem(item)
        }
    }
    if (!isValidItem(item)) {
        throw "Not a standard item"
    }
    return item;
}

/**
 * Checks if the item is a valid item.           
 * @param {Item} item - the item to check           
 * @returns {boolean} - true if the item is a valid item, false otherwise           
 */
function isValidItem(item) {
    return !isUndefined(item) && isStandardItem(item) && !isEmpty(item.label.trim())
}

/**
 * Takes in an object and converts it to a standard item.           
 * @param {object} item - the object to convert.           
 * @returns {StandardItem} - the converted object.
 * @throws {InvalidItemException} - When item has no label property or the label property is an empty string           
 */
function convertObjectToStandardItem(item) {
    let title = Object.keys(item)[0]
    let value = item[title];
    let result = {
        label: title,
        items: isArray(value) ? value.map(e => buildStandardItem(e)) : buildStandardItem(value)
    }

    if (!isValidItem(result)) {
        throw new InvalidItemException("Not standard item");
    }
    return result
}

/**
 * Checks if the given item is a standard item.           
 * @param {object} item - the item to check           
 * @returns {boolean} - true if the item is a standard item, false otherwise           
 */
function isStandardItem(item) {
    return ['label'].every(key => Object.keys(item).includes(key))
}

/**
 * An exception that is thrown when an item is not a SidebarItem.           
 * @param {string} message - the message to display when the exception is thrown.           
 */
class InvalidItemException extends Error {
    constructor(message) {
        super('Not a SidebarItem item')
    }
}
